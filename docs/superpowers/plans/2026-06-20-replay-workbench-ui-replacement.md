# Replay Workbench UI Replacement Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the current basic Sentinel Archive React control panel with the approved chart-first Replay Workbench while preserving the existing FastAPI API and launcher behavior.

**Architecture:** Keep the backend contract stable and rebuild the frontend as focused React modules under `frontend/src/replayWorkbench/`. Pure derivation logic is tested with Vitest, the custom replay chart is built from SVG/HTML with no new chart dependency, and the existing API client remains the single boundary to FastAPI.

**Tech Stack:** React 19, Vite 8, TypeScript 6, lucide-react, FastAPI endpoints already exposed by `sentinel_archive/api.py` and `sentinel_archive/recorder_api.py`, Vitest for frontend utility tests, Playwright for browser verification.

```ts
import type { ParsedAlert, PriceDriftEvent, ReplaySession, SimulationSnapshot } from '../api';
import type { BotLane, EventTapeRow, FillConfidence } from './types';

export function formatMoney(value: unknown) {
  const parsed = Number(value);
  if (!Number.isFinite(parsed)) return 'Unavailable';
  return new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD', maximumFractionDigits: 2 }).format(parsed);
}

export function formatNumber(value: unknown, digits = 2) {
  const parsed = Number(value);
  if (!Number.isFinite(parsed)) return 'Unavailable';
  return new Intl.NumberFormat('en-US', { maximumFractionDigits: digits }).format(parsed);
}

export function selectedReplaySession(snapshot: SimulationSnapshot | null, selectedSessionId: string) {
  if (!snapshot?.sessions.length) return null;
  return snapshot.sessions.find((session) => session.session_id === selectedSessionId) ?? snapshot.sessions[0] ?? null;
}

export function replayProgress(snapshot: SimulationSnapshot | null, session: ReplaySession | null) {
  const current = Math.max(0, Number(snapshot?.replay.index ?? 0));
  const total = Math.max(0, Number(session?.bar_count ?? 0));
  const percent = total > 0 ? Math.min(100, Math.round((current / total) * 100)) : 0;
  return { current, total, percent };
}

export function fillConfidenceForSession(session: ReplaySession | null): FillConfidence {
  if (!session) {
    return { label: 'Low', tone: 'bad', reason: 'No replay session is selected.' };
  }
  if (session.bar_count <= 0 || session.symbols.length === 0) {
    return { label: 'Low', tone: 'bad', reason: 'Session metadata is missing bars or symbols.' };
  }
  return { label: 'Medium', tone: 'warn', reason: 'OHLCV bars with configured slippage, commission, and high/low stop checks.' };
}

export function buildBotLanes(snapshot: SimulationSnapshot | null, alerts: ParsedAlert[], driftEvents: PriceDriftEvent[]): BotLane[] {
  const active = Boolean(snapshot?.replay.active);
  const openPositions = Number(snapshot?.account.open_positions ?? 0);
  const decisions = snapshot?.decisions.length ?? 0;
  const parsedAlerts = alerts.filter((alert) => alert.parse_status === 'parsed').length;
  const driftAlerts = driftEvents.filter((event) => event.price_drift_alert).length;
  return [
    {
      id: 'pulse',
      title: 'Sentinel Pulse',
      status: active ? 'Replay connected' : 'Waiting for replay',
      detail: `${openPositions} open simulated position${openPositions === 1 ? '' : 's'}`,
      tone: active ? 'good' : 'neutral',
    },
    {
      id: 'edge',
      title: 'Sentinel Edge',
      status: decisions ? 'Decision feed active' : 'No decisions yet',
      detail: `${decisions} replay decision${decisions === 1 ? '' : 's'} observed`,
      tone: decisions ? 'good' : 'neutral',
    },
    {
      id: 'sentinel-echo',
      title: 'Sentinel Echo',
      status: parsedAlerts ? 'Replay events ready' : 'No parsed alerts',
      detail: `${parsedAlerts} parsed Discord alert${parsedAlerts === 1 ? '' : 's'}`,
      tone: parsedAlerts ? 'good' : 'neutral',
    },
    {
      id: 'crypto',
      title: 'Sentinel-Chain',
      status: active ? 'Paper path available' : 'Awaiting price path',
      detail: 'Uses replay prices for bracket-lot stop and target review.',
      tone: active ? 'good' : 'neutral',
    },
    {
      id: 'darkpool',
      title: 'Sentinel-Flare',
      status: driftAlerts ? 'Manual review flagged' : 'Manual review lane',
      detail: driftAlerts ? `${driftAlerts} drift event${driftAlerts === 1 ? '' : 's'} need review` : 'Intent packets stay review-only.',
      tone: 'warn',
    },
  ];
}

export function buildEventTape(snapshot: SimulationSnapshot | null, alerts: ParsedAlert[], driftEvents: PriceDriftEvent[]): EventTapeRow[] {
  const rows: EventTapeRow[] = [];
  if (snapshot?.replay.current_timestamp) {
    rows.push({
      id: `replay-${snapshot.replay.current_timestamp}`,
      timestamp: snapshot.replay.current_timestamp,
      source: 'Replay',
      message: `Replay cursor at index ${snapshot.replay.index}`,
      tone: snapshot.replay.active ? 'good' : 'neutral',
    });
  }
  for (const event of snapshot?.event_log ?? []) {
    const timestamp = typeof event.timestamp === 'number' ? new Date(event.timestamp * 1000).toISOString() : String(event.timestamp ?? 'event');
    rows.push({
      id: `engine-${timestamp}-${String(event.event_type ?? 'event')}`,
      timestamp,
      source: 'Engine',
      message: [event.event_type, event.symbol, event.status].filter(Boolean).join(' · '),
      tone: String(event.status ?? '').includes('reject') ? 'bad' : 'neutral',
    });
  }
  for (const alert of alerts.slice(0, 8)) {
    rows.push({
      id: `alert-${alert.message_id}`,
      timestamp: alert.message_id,
      source: 'Discord',
      message: alert.parse_status === 'parsed' ? [alert.action, alert.ticker, alert.strike, alert.option_type].filter(Boolean).join(' ') : alert.raw_text,
      tone: alert.parse_status === 'parsed' ? 'good' : 'warn',
    });
  }
  for (const drift of driftEvents.slice(0, 8)) {
    rows.push({
      id: `drift-${drift.alert_id}`,
      timestamp: drift.alert_id,
      source: 'Drift',
      message: `Alert ${drift.alert_id} drift ${formatNumber(drift.price_drift_pct)}%`,
      tone: drift.price_drift_alert ? 'warn' : 'neutral',
    });
  }
  return rows.slice(0, 40);
}
```

```ts
import type { MarketBar } from '../api';

export type ChartDomain = { min: number; max: number };
export type VolumeBar = { x: number; height: number; tone: 'up' | 'down' };

export function visibleBarsForSymbol(bars: MarketBar[], symbol: string) {
  const normalized = symbol.trim().toUpperCase();
  return bars.filter((bar) => bar.symbol.toUpperCase() === normalized);
}

export function chartDomain(bars: MarketBar[]): ChartDomain {
  if (!bars.length) return { min: 0, max: 1 };
  const lows = bars.map((bar) => Number(bar.low)).filter(Number.isFinite);
  const highs = bars.map((bar) => Number(bar.high)).filter(Number.isFinite);
  const min = Math.min(...lows);
  const max = Math.max(...highs);
  if (min === max) return { min: min - 1, max: max + 1 };
  const pad = (max - min) * 0.05;
  return { min: round2(min - pad), max: round2(max + pad) };
}

export function scaleY(value: number, domain: ChartDomain, height: number) {
  const span = domain.max - domain.min || 1;
  return round2(height - ((value - domain.min) / span) * height);
}

export function buildLinePath(bars: MarketBar[], width: number, height: number) {
  if (!bars.length) return '';
  const domain = chartDomain(bars);
  const lastIndex = Math.max(1, bars.length - 1);
  return bars
    .map((bar, index) => {
      const x = round2((index / lastIndex) * width);
      const y = scaleY(bar.close, domain, height);
      return `${index === 0 ? 'M' : 'L'} ${x} ${y}`;
    })
    .join(' ');
}

export function buildVolumeBars(bars: MarketBar[], maxHeight: number): VolumeBar[] {
  const maxVolume = Math.max(1, ...bars.map((bar) => Number(bar.volume) || 0));
  return bars.map((bar, index) => ({
    x: index,
    height: ((Number(bar.volume) || 0) / maxVolume) * maxHeight,
    tone: bar.close >= bar.open ? 'up' : 'down',
  }));
}

function round2(value: number) {
  return Math.round(value * 100) / 100;
}
```
